*****CREACION TABLA PAGOSGYM*****
CREATE TABLE PAGOSGYM (
id_Pago NUMBER GENERATED by default on null as IDENTITY,
CONSTRAINT pkPagoGYM PRIMARY KEY (id_Pago),
fecha_Pago date,
Cantidad float,
mes_Pagado varchar(15),
no_Cliente int,
CONSTRAINT fkPagoCliente FOREIGN KEY (no_Cliente) REFERENCES CLIENTESGYM(id_Cliente)
);
********************************


*****CREACION TABLA CLIENTESGYM*****
CREATE TABLE CLIENTESGYM(
id_Cliente NUMBER GENERATED by default on null as IDENTITY,
CONSTRAINT pkClienteGYM PRIMARY KEY (id_Cliente),
nombre varchar(35),
apellidos varchar(35),
telefono varchar(8),
DPI varchar(13),
fecha_Inscripcion date
);
